9pub fn derive_binary_serde(input_tokens: proc_macro::TokenStream) -> proc_macro::TokenStream {
10    let input = parse_macro_input!(input_tokens as DeriveInput);
11    match &input.data {
12        syn::Data::Struct(data_struct) => {
13            let field_types = data_struct
14                .fields
15                .iter()
16                .map(|field| TypeExpr::from_type(&field.ty));
17
18            gen_impl(GenImplParams {
19                type_ident: input.ident,
20                generics: input.generics,
21                additional_where_predicates: gen_predicates_for_field_types(field_types.clone())
22                    .collect(),
23                serialized_size: field_types
24                    .clone()
25                    .map(|field_type| field_type.serialized_size())
26                    .sum(),
27                recursive_array_type: gen_recursive_array_type_for_field_types(field_types),
28                serialization_code: gen_struct_serialization_code(&data_struct.fields),
29                deserialization_code: gen_struct_deserialization_code(&data_struct.fields),
30            })
31            .into()
32        }
33        syn::Data::Enum(data_enum) => {
34            // make sure that none of the enum's variants hold data in them
35            for variant in &data_enum.variants {
36                if !matches!(&variant.fields, syn::Fields::Unit) {
37                    return quote_spanned! {
38                        proc_macro2::Span::call_site() => compile_error!("enum variants which contain data are not supported");
39                    }.into();
40                }
41            }
42            let repr_type = match enum_get_repr_type(&input) {
43                Ok(repr_type) => TypeExpr(repr_type.clone()),
44                Err(err) => return err.into(),
45            };
46            gen_impl(GenImplParams {
47                additional_where_predicates: Vec::new(),
48                serialized_size: repr_type.serialized_size(),
49                recursive_array_type: repr_type.serialized_recursive_array_type(),
50                serialization_code: quote! {
51                    let as_primitive: &#repr_type = unsafe { ::core::mem::transmute(self) };
52                    ::binary_serde::BinarySerde::binary_serialize(as_primitive, buf, endianness)
53                },
54                deserialization_code: gen_enum_deserialization_code(
55                    &input.ident,
56                    repr_type,
57                    data_enum,
58                ),
59                type_ident: input.ident,
60                generics: input.generics,
61            })
62            .into()
63        }
64        syn::Data::Union(_) => {
65            return quote_spanned! {
66                proc_macro2::Span::call_site() => compile_error!("unions are not supported, only structs are supported");
67            }
68            .into();
69        }
70    }
71}

73/// attempts to extract the enum's underlying representation type.
74fn enum_get_repr_type<'a>(
75    derive_input: &'a DeriveInput,
76) -> Result<&'a proc_macro2::TokenStream, proc_macro2::TokenStream> {
77    let Some(repr_type) = derive_input.attrs.iter().find_map(|attr| {
78        let syn::AttrStyle::Outer = &attr.style else {
79            return None;
80        };
81        let syn::Meta::List(meta_list) = &attr.meta else {
82            return None;
83        };
84        if !meta_list.path.is_ident("repr") {
85            return None;
86        }
87        Some(&meta_list.tokens)
88    }) else {
89        return Err(quote_spanned! {
90            proc_macro2::Span::call_site() => compile_error!("a #[repr(...)] attribute is required on the enum to specify the size of the enum's tag");
91        });
92    };
93
94    if !is_enum_repr_type_sized_primitive_int(&repr_type) {
95        return Err(quote_spanned! {
96            proc_macro2::Span::call_site() => compile_error!("the enum's #[repr(...)] attribute must contain an explicitly sized primitive integer type (e.g., u32)");
97        });
98    }
99
100    Ok(repr_type)
101}

103/// checks if the provided expression inside a `#[repr(...)]` attribute on an enum represents an explicitly sized primitive int.
104fn is_enum_repr_type_sized_primitive_int(repr_type: &proc_macro2::TokenStream) -> bool {
105    let repr_type_str = repr_type.to_string();
106    repr_type_str.as_bytes()[0].is_ascii_alphabetic() && repr_type_str[1..].parse::<u32>().is_ok()
107}

109/// returns an iterator over the offset and size of each field.
110fn fields_offsets_and_sizes<'a, I: Iterator<Item = &'a syn::Field> + 'a>(
111    fields: I,
112) -> impl Iterator<Item = FieldOffsetAndSize> + 'a {
113    fields.scan(SerializedSizeExpr::zero(), |prev_fields_size, cur_field| {
114        let cur_field_size = TypeExpr::from_type(&cur_field.ty).serialized_size();
115        let new_size = &*prev_fields_size + &cur_field_size;
116
117        // the offset of this field is the size of all previous fields, and update the prev size to the new size.
118        let offset = core::mem::replace(prev_fields_size, new_size);
119
120        Some(FieldOffsetAndSize {
121            size: cur_field_size,
122            offset: FieldOffsetExpr(offset.0),
123        })
124    })
125}
126
127/// information about the offset and size of a field.
128struct FieldOffsetAndSize {
129    size: SerializedSizeExpr,
130    offset: FieldOffsetExpr,
131}

133/// generates accessors for the given fields. each accessor can be prepended with `self.` to access the value of the field.
134fn field_accessors<'a, I: Iterator<Item = &'a syn::Field> + 'a>(
135    fields: I,
136) -> impl Iterator<Item = syn::Member> + 'a {
137    fields
138        .enumerate()
139        .map(|(field_index, field)| match &field.ident {
140            Some(ident) => syn::Member::Named(ident.clone()),
141            None => syn::Member::Unnamed(syn::Index::from(field_index)),
142        })
143}

145/// generates code for serializing a struct with the given fields.
146fn gen_struct_serialization_code(fields: &syn::Fields) -> proc_macro2::TokenStream {
147    let statements = field_accessors(fields.iter())
148        .zip(fields_offsets_and_sizes(fields.iter()))
149        .map(|(field_accessor, offset_and_size)| {
150            let FieldOffsetAndSize { size, offset } = offset_and_size;
151            quote! {
152                ::binary_serde::BinarySerde::binary_serialize(
153                    &self.#field_accessor,
154                    &mut buf[#offset..][..#size],
155                    endianness
156                );
157            }
158        });
159    quote! {
160        #(#statements)*
161    }
162}

164/// generates code for deserializing a struct with the given fields.
165fn gen_enum_deserialization_code(
166    enum_ident: &syn::Ident,
167    enum_repr_type: TypeExpr,
168    data_enum: &syn::DataEnum,
169) -> proc_macro2::TokenStream {
170    let variant_consts_idents = data_enum.variants.iter().map(|variant| {
171        let const_name = format!(
172            "_BINARY_SERDE_CONST_{}",
173            pascal_to_capital_snake_case(&variant.ident.to_string())
174        );
175        syn::Ident::new(&const_name, proc_macro2::Span::mixed_site())
176    });
177    let variant_consts_definitions = data_enum
178        .variants
179        .iter()
180        .zip(variant_consts_idents.clone())
181        .map(|(variant, const_ident)| {
182            let variant_ident = &variant.ident;
183            quote! {
184                #[allow(non_upper_case_globals)]
185                const #const_ident: #enum_repr_type = (#enum_ident::#variant_ident) as #enum_repr_type;
186            }
187        });
188    let match_cases =
189        data_enum
190            .variants
191            .iter()
192            .zip(variant_consts_idents)
193            .map(|(variant, const_ident)| {
194                let variant_ident = &variant.ident;
195                quote! {
196                    #const_ident => Ok(#enum_ident::#variant_ident)
197                }
198            });
199    let enum_name = enum_ident.to_string();
200    quote! {
201        #(#variant_consts_definitions)*
202        let primitive_value = <#enum_repr_type as ::binary_serde::BinarySerde>::binary_deserialize(buf, endianness)?;
203        match primitive_value {
204            #(#match_cases,)*
205            _ => ::core::result::Result::Err(::binary_serde::DeserializeError::InvalidEnumValue {
206                enum_name: #enum_name,
207            })
208        }
209    }
210}

212/// convets a PascalCase string to a CAPITAL_SNAKE_CASE string.
213fn pascal_to_capital_snake_case(s: &str) -> String {
214    let mut result = String::new();
215    for (i, c) in s.chars().enumerate() {
216        if c.is_uppercase() && i != 0 {
217            result.push('_');
218        }
219        result.extend(c.to_uppercase())
220    }
221    result
222}

224/// generates code for deserializing a struct with the given fields.
225fn gen_struct_deserialization_code(fields: &syn::Fields) -> proc_macro2::TokenStream {
226    let field_values = fields_offsets_and_sizes(fields.iter()).map(|offset_and_size| {
227        let FieldOffsetAndSize { size, offset } = offset_and_size;
228        quote! {
229            ::binary_serde::BinarySerde::binary_deserialize(
230                &buf[#offset..][..#size],
231                endianness
232            )?
233        }
234    });
235    match fields {
236        syn::Fields::Named(_) => {
237            let field_specifications =
238                fields.iter().zip(field_values).map(|(field, field_value)| {
239                    let field_ident = field.ident.as_ref().unwrap();
240                    quote! {
241                        #field_ident: #field_value
242                    }
243                });
244            quote! {
245                Ok(Self {
246                    #(#field_specifications),*
247                })
248            }
249        }
250        syn::Fields::Unnamed(_) => {
251            quote! {
252                Ok(Self (
253                    #(#field_values),*
254                ))
255            }
256        }
257        syn::Fields::Unit => quote! { Ok(Self) },
258    }
259}
260/// generates a recursive array type for a struct made of the given field types.
261fn gen_recursive_array_type_for_field_types<I: Iterator<Item = TypeExpr>>(
262    field_types: I,
263) -> TypeExpr {
264    let empty_array = quote! {::binary_serde::recursive_array::EmptyRecursiveArray};
265    let final_array_expr = field_types.fold(empty_array, |cur_array_type, cur_field_type| {
266        let cur_field_recursive_array_type = cur_field_type.serialized_recursive_array_type();
267        quote! {
268            ::binary_serde::recursive_array::RecursiveArrayConcatenation<u8, #cur_array_type, #cur_field_recursive_array_type>
269        }
270    });
271    TypeExpr(final_array_expr)
